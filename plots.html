#uploading data
noaa = read.csv("nynoaadat.csv") %>%
  mutate(date = as.Date(date))

noaa_long_10_20 = noaa %>%
  mutate(date = as.Date(date)) %>%
  filter(year(date) >= 2010, year(date) <= 2020) %>%
  mutate(month_num = month(date, label = TRUE))

noaa_wide_10_20 = noaa_long_10_20 %>%
  mutate(month = floor_date(date, "month")) %>%
  group_by(id, month) %>%
  summarise(median_tmax = median(tmax, na.rm = TRUE), .groups = "drop") %>%
  pivot_wider(names_from = month,
    values_from = median_tmax,
    names_prefix = "median_")
